Алгоритм не проходит по внешним ссылкам. На сайте собираются все станицы, которые имеют в своем адресе строк «/tag». Результат сохраняется в csv таблице, где первый столбик, порядковый номер, второй — заголовок страницы, третий — полный адрес страницы
 

### Получаем в ответ код

~~~python
	import requests
	from bs4 import BeautifulSoup
	from urllib.parse import urljoin, urlparse
	import csv
	import time
	from collections import deque
	
	class WebsiteParser:
	    def __init__(self, base_url):
	        self.base_url = base_url.rstrip('/')
	        self.parsed_urls = set()
	        self.tag_pages = []
	        self.session = requests.Session()
	        self.session.headers.update({
	            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
	        })
	    
	    def is_internal_link(self, url):
	        """Проверяет, является ли ссылка внутренней"""
	        parsed_base = urlparse(self.base_url)
	        parsed_url = urlparse(url)
	        
	        # Проверяем, что домены совпадают или ссылка относительная
	        return (not parsed_url.netloc or 
	                parsed_url.netloc == parsed_base.netloc)
	    
	    def is_tag_page(self, url):
	        """Проверяет, содержит ли URL строку /tag"""
	        return '/tag' in url
	    
	    def get_page_content(self, url):
	        """Получает содержимое страницы"""
	        try:
	            response = self.session.get(url, timeout=10)
	            response.raise_for_status()
	            return response.text
	        except requests.RequestException as e:
	            print(f"Ошибка при загрузке {url}: {e}")
	            return None
	    
	    def extract_links(self, html, base_url):
	        """Извлекает все ссылки со страницы"""
	        if not html:
	            return []
	        
	        soup = BeautifulSoup(html, 'html.parser')
	        links = []
	        
	        for link in soup.find_all('a', href=True):
	            href = link.get('href')
	            if href:
	                full_url = urljoin(base_url, href)
	                links.append(full_url)
	        
	        return links
	    
	    def parse_website(self):
	        """Основной метод парсинга сайта"""
	        print(f"Начинаем парсинг сайта: {self.base_url}")
	        
	        # Используем очередь для обхода в ширину
	        queue = deque([self.base_url])
	        
	        while queue:
	            current_url = queue.popleft()
	            
	            # Пропускаем уже обработанные URL
	            if current_url in self.parsed_urls:
	                continue
	                
	            self.parsed_urls.add(current_url)
	            print(f"Обрабатываем: {current_url}")
	            
	            # Получаем содержимое страницы
	            html = self.get_page_content(current_url)
	            if not html:
	                continue
	            
	            # Парсим заголовок страницы
	            soup = BeautifulSoup(html, 'html.parser')
	            title = soup.find('title')
	            page_title = title.get_text().strip() if title else "Без заголовка"
	            
	            # Если это страница с тегом, сохраняем информацию
	            if self.is_tag_page(current_url):
	                self.tag_pages.append({
	                    'url': current_url,
	                    'title': page_title
	                })
	                print(f"Найдена страница с тегом: {page_title}")
	            
	            # Извлекаем все ссылки со страницы
	            links = self.extract_links(html, current_url)
	            
	            # Добавляем внутренние ссылки в очередь
	            for link in links:
	                if (self.is_internal_link(link) and 
	                    link not in self.parsed_urls and 
	                    link not in queue):
	                    queue.append(link)
	            
	            # Задержка между запросами
	            time.sleep(1)
	        
	        print(f"Парсинг завершен. Найдено страниц с тегами: {len(self.tag_pages)}")
	    
	    def save_to_csv(self, filename):
	        """Сохраняет результаты в CSV файл"""
	        with open(filename, 'w', newline='', encoding='utf-8') as csvfile:
	            fieldnames = ['№', 'Заголовок', 'URL']
	            writer = csv.DictWriter(csvfile, fieldnames=fieldnames)
	            
	            writer.writeheader()
	            for i, page in enumerate(self.tag_pages, 1):
	                writer.writerow({
	                    '№': i,
	                    'Заголовок': page['title'],
	                    'URL': page['url']
	                })
	        
	        print(f"Результаты сохранены в файл: {filename}")
	
	def main():
	    # Получаем URL от пользователя
	    url = input("Введите URL сайта (например, https://sdelano.media): ").strip()
	    
	    if not url.startswith(('http://', 'https://')):
	        url = 'https://' + url
	    
	    # Создаем парсер и запускаем
	    parser = WebsiteParser(url)
	    parser.parse_website()
	    
	    # Сохраняем результаты
	    if parser.tag_pages:
	        filename = f"tag_pages_{urlparse(url).netloc}.csv"
	        parser.save_to_csv(filename)
	    else:
	        print("Страницы с тегами не найдены.")
	
	if __name__ == "__main__":
	    main()
    ~~~
